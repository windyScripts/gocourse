package main

import (
	"flag"
	"fmt"
	"os"
)

func main(){
 fmt.Println("command:", os.Args[0])

 for i, arg := range os.Args {
	fmt.Println("Argument", i, ":", arg)
 }

 //fmt.Println("arg1", os.Args[1]) //useful for cli applications.

 var name string
 var age int
 var male bool

 flag.StringVar(&name, "name", "John", "Name of the user.")
 flag.IntVar(&age, "age", 18, "Age of the user.")
 flag.BoolVar(&male, "male", true, "Is the user male")

 flag.Parse()

 fmt.Println("Name", name)
 fmt.Println("Age", age)
 fmt.Println("Male", male)

 //"go run 55_CLargumentsFlags.go -name Jane Doe -age 20 -male false"
 // sanitize user input before use
}

package main

import (
	"fmt"
	"log"
	"net/http"
)

func main() {

	http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request){
		fmt.Fprintln(w, "Hello Server!") //writes to w.
	})


	//const serverAddr string = "127.0.0.1:3000"
	const port string = ":3000" // assumes localhost.
	// receives requests, sends responses.
	err := http.ListenAndServe(/* serverAddr */port, nil)
	if err != nil {
		log.Fatalln("error starting server", err)
		// 65535 ports in total.
	}
	
}
